sudo tee /usr/local/sbin/restore_milestone4 > /dev/null << 'EOF'
#!/bin/bash
set -euo pipefail

echo "=== Milestone 4 Restore Starting ==="

echo "Stopping scheduler..."
systemctl stop intra-hub-sync.timer 2>/dev/null || true
systemctl stop intra-hub-sync.service 2>/dev/null || true

APP=$(ls -t /opt/intra-hub-v1.0/backups/intra-hub_FULL_STABLE_EXCELLENT_CLEAN_*.tgz 2>/dev/null | head -n 1 || true)
if [[ -z "${APP}" ]]; then
  echo "ERROR: No CLEAN backup found."
  exit 1
fi

CFG=$(ls -t /opt/intra-hub-v1.0/backups/intra-hub_SYSTEM_CONFIG_*.tgz 2>/dev/null | head -n 1 || true)

echo "Using APP backup: $APP"
echo "Using CFG backup: ${CFG:-<none>}"

if [[ -n "${CFG}" ]]; then
  echo "Restoring system config..."
  tar xzf "${CFG}" -C /
fi

echo "Restoring application..."
rm -rf /opt/intra-hub-v1.0
tar xzf "${APP}" -C /opt

echo "Fixing permissions..."
chown -R www-data:www-data /opt/intra-hub-v1.0/{public,data,logs,backups} 2>/dev/null || true
chown root:www-data /opt/intra-hub-v1.0/.env 2>/dev/null || true
chmod 640 /opt/intra-hub-v1.0/.env 2>/dev/null || true

echo "Reloading services..."
systemctl daemon-reload
nginx -t
systemctl reload nginx || systemctl restart nginx

echo "Starting scheduler..."
systemctl start intra-hub-sync.timer

echo "=== Milestone 4 restored successfully ==="
EOF

sudo chmod +x /usr/local/sbin/restore_milestone4
